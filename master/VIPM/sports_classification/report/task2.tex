\section{Task 2}
\makeatletter\def\@currentlabel{Task 2}\makeatother\label{sec:task2}

\subsection{Overview}
\begin{frame}{Overview}
  \begin{block}{Prompt}
    Classify the degraded test set using two non-exclusive approaches:
    unsupervised image enhancement at inference time, and
    augmentation-aware retraining of the classifier.
  \end{block}
  \vspace{\baselineskip}
  The degraded test set contains 500 images sharing the same ground truth as the clean test set,
  but acquired under strong degradation.
  Our workflow was to first measure the performance drop, then characterize the degradation.
\end{frame}

\subsection{Baseline on the Degraded Set}
\begin{frame}{Baseline on the Degraded Set}
  We began by evaluating the Task 1 best model (83.6\% clean accuracy)
  directly on the degraded test set without any modification.
  \vspace{\baselineskip}
  \begin{itemize}
    \item [clean] \textit{Accuracy}: 83.6\%, \textit{F1}: 0.827
    \item [degraded] \textit{Accuracy}: 56.2\%, \textit{F1}: 0.554
  \end{itemize}
  \vspace{0.5\baselineskip}
  A \textbf{27-point drop}: the model had never seen degraded images during training
  and could not generalize to them.
\end{frame}

\subsection{Approach A \textendash\ Unsupervised Enhancement}
\begin{frame}{Unsupervised Enhancement \textendash\ Techniques}
  Before modifying training, we attempted to correct the degradation \textit{at inference time}
  using classic unsupervised techniques:
  \vspace{0.5\baselineskip}
  \begin{itemize}
    \item \textbf{Auto Gamma Correction}: estimates a per-image gamma exponent from the mean
          brightness to pull it towards midgray (0.5).
          Dark images are brightened ($\gamma < 1$), bright images darkened ($\gamma > 1$).
    \item \textbf{White Balance --- Gray World}: assumes the average scene color is neutral grey;
          scales each RGB channel by $0.5\,/\,\mu_c$.
    \item \textbf{Contrast Stretch}: expands the dynamic range using the 8\textsuperscript{th}
          percentile as the black point, clipping and rescaling to $[0, 1]$.
  \end{itemize}
\end{frame}

\begin{frame}{Unsupervised Enhancement \textendash\ Results}
  \begin{center}
    \begin{tabular}{lcc}
      \toprule
      Technique & Accuracy & F1 \\
      \midrule
      No enhancement (baseline) & 56.2\% & 0.554 \\
      Auto Gamma Correction     & 47.6\% & 0.462 \\
      White Balance (Gray World) & 53.2\% & 0.530 \\
      WB + Gamma                & 45.0\% & 0.446 \\
      Contrast Stretch          & $\approx$52\% & --- \\
      \bottomrule
    \end{tabular}
  \end{center}
  \vspace{0.5\baselineskip}
  Every technique worsened or negligibly changed the results.
  These methods all act on \textit{brightness}, \textit{contrast} and \textit{white balance}
  --- yet none of them improved accuracy. \\
  \vspace{0.5\baselineskip}
  $\Rightarrow$ The degradation must be different in nature.
  We decided to understand what it actually is before going further.
\end{frame}

\subsection{Degraded Set Analysis}
\begin{frame}{Degraded Set Analysis \textendash\ Statistics}
  We performed a \textbf{paired pixel-level comparison} of all 500 clean/degraded image pairs
  to measure the degradation quantitatively.
  \vspace{0.5\baselineskip}
  \begin{center}
    \begin{tabular}{lccc}
      \toprule
      Metric & Clean & Degraded & $\Delta$ \\
      \midrule
      Brightness (mean pixel, 0--255) & 118.8 & 118.9 & $+$0.15 \\
      Contrast (pixel std) & 61.4 & 62.6 & $+$1.19 \\
      \textbf{Saturation (HSV, 0--255)} & 83.1 & \textbf{100.6} & \textbf{$+$17.5} \\
      Hue std & 60.0 & 52.8 & $-$7.2 \\
      \bottomrule
    \end{tabular}
  \end{center}
  \vspace{0.5\baselineskip}
  Brightness and contrast are \textit{statistically neutral}. \textbf{Saturation} is the
  primary degradation: 66\% of images are more saturated, with a mean ratio of
  \textbf{1.65$\times$} and peaks up to 7.11$\times$.

  \textbf{Hue shift} is the secondary degradation: 78\% of images show a shift larger than
  14°, with a standard deviation of 0.177 in torchvision units --- requiring
  \texttt{hue}$\approx$0.25 to cover 80\% of cases.
\end{frame}

\subsection{Approach B \textendash\ Augmentation-Aware Training}
\begin{frame}{Iteration v2.0 \textendash\ First ColorJitter}
  We retrained the model from scratch adding \textbf{ColorJitter} to the training pipeline
  to simulate color distortions at training time.
  \vspace{\baselineskip}
  \begin{columns}[onlytextwidth]
    \begin{column}{0.35\linewidth}
      \begin{itemize}
        \item [train] Loss: 1.2886,\\ Acc: 0.6409
        \item [val] Loss: 0.9709,\\ Acc: 0.7460
        \item [clean] \textit{Acc}: 77.2\%,\\ \textit{F1}: 0.761
        \item [degraded] \textit{Acc}: 58.6\%,\\ \textit{F1}: 0.579
      \end{itemize}
    \end{column}
    \begin{column}{0.65\linewidth}
      \begin{center}
        \includegraphics[width=\textwidth]{../models/v2.0_color_jitter.png}
      \end{center}
    \end{column}
  \end{columns}
  \vspace{0.5\baselineskip}
  Degraded accuracy improved from 56.2\% to 58.6\%.
  Color augmentation is the right direction.
  The clean set drops slightly as the model must now handle color variation.

  \let\thefootnote\relax\footnotetext{Trained on RX9070-WSL \textendash\ 32mins 23secs}
\end{frame}

\begin{frame}{Iteration v2.1 \textendash\ More Epochs}
  We extended training to \textbf{80 epochs} to let the model fully converge
  under color variation.
  \vspace{\baselineskip}
  \begin{columns}[onlytextwidth]
    \begin{column}{0.35\linewidth}
      \begin{itemize}
        \item [train] Loss: 1.0131,\\ Acc: 0.7131
        \item [val] Loss: 0.8720,\\ Acc: 0.7600
        \item [clean] \textit{Acc}: 78.5\%,\\ \textit{F1}: 0.775
        \item [degraded] \textit{Acc}: 62.0\%,\\ \textit{F1}: 0.611
      \end{itemize}
    \end{column}
    \begin{column}{0.65\linewidth}
      \begin{center}
        \includegraphics[width=\textwidth]{../models/v2.1_more_epochs.png}
      \end{center}
    \end{column}
  \end{columns}
  \vspace{0.5\baselineskip}
  Small consistent improvement from longer training.
  We next focused on whether the augmentation parameters themselves were well-calibrated
  for the actual degradation.
  \let\thefootnote\relax\footnotetext{Trained on RX9070-WSL \textendash\ 39mins 57secs}
\end{frame}

\begin{frame}[fragile]{Iteration v2.2 \textendash\ Jitter Tuning (Wrong Direction)}
  We adjusted ColorJitter to protect \textit{key color features},
  which are strong discriminative cues (e.g.\ field and sky colors):
  \begin{Verbatim}[fontsize=\footnotesize]
ColorJitter(brightness=0.3, contrast=0.3,
            saturation=0.2, hue=0.05)
  \end{Verbatim}
  \texttt{hue} was reduced from 0.1 to 0.05 (covering only $\pm$18°) and saturation was kept conservatively low.

  \vspace{-0.6\baselineskip}
  \begin{columns}[onlytextwidth]
    \begin{column}{0.3\linewidth}
      \begin{itemize}
        \item [clean] \textit{Acc}: 85.1\%,\\ \textit{F1}: 0.843
        \item [degraded] \textit{Acc}: 66.9\%,\\ \textit{F1}: 0.668
      \end{itemize}
    \end{column}
    \begin{column}{0.7\linewidth}
      \begin{center}
        \includegraphics[width=\textwidth]{../models/v2.2_jitter_param_tuning.png}
      \end{center}
    \end{column}
  \end{columns}
  Clean accuracy improved strongly, but \texttt{hue=0.05} covers only $\pm$18° while
  78\% of degraded images have shifts \texttt{\char`<}14° \textendash{} the parameter is \textbf{4 \texttimes{} too small}.
  \let\thefootnote\relax\footnotetext{Trained on RX6950 \textendash\ 19mins 20secs}
\end{frame}

\begin{frame}[fragile]{Iteration v2.3 \textendash\ Analysis-Driven Parameters}
  We re-derived the ColorJitter parameters directly from the degraded set analysis:
  \begin{Verbatim}[fontsize=\footnotesize]
ColorJitter(brightness=0.35, contrast=0.4,
            saturation=(0.7, 3.0), hue=0.25)
  \end{Verbatim}
  \begin{itemize}
    \item \texttt{hue=0.25}: covers 80\% of the observed hue shifts (std 0.177 in torchvision units)
    \item \texttt{saturation=(0.7, 3.0)}: asymmetric tuple --- the degradation is
          almost entirely a saturation \textit{increase} (mean 1.65$\times$, max 7.11$\times$)
  \end{itemize}

  \vspace{-0.6\baselineskip}
  \begin{columns}[onlytextwidth]
    \begin{column}{0.3\linewidth}
      \begin{itemize}
        \item [clean] \textit{Acc}: 84.3\%,\\ \textit{F1}: 0.838
        \item [degraded] \textit{Acc}: 73.1\%,\\ \textit{F1}: 0.727
      \end{itemize}
    \end{column}
    \begin{column}{0.7\linewidth}
      \begin{center}
        \includegraphics[width=\textwidth]{../models/v2.3_jitter_param_tuning.png}
      \end{center}
    \end{column}
  \end{columns}
  \textbf{$+$6.2\%} on the degraded set over v2.2
\end{frame}

\begin{frame}{Failed Experiments}
  After v2.3 we explored additional augmentation strategies to close the remaining gap.
  All were \textbf{reverted}:
  \vspace{0.5\baselineskip}
  \begin{center}
    \resizebox{\textwidth}{!}{%
    \begin{tabular}{lcc}
      \toprule
      Iteration & Degraded Acc & Why reverted \\
      \midrule
      v2.4 --- Gaussian noise & 60.1\% & The majority of edge features became irrelevant \\
      v2.5 --- JPEG compression & 62.8\% & In the dataset we didn't find any JPEG compression noise \\
      v2.6 --- More hue shift & 66.2\% & Exceeds data range, introduces more class confusion \\
      v2.8 --- Mixup ($\alpha=0.2$) & 46.2\% & Too much regularisation; CNN underfits \\
      v2.9 --- Mixup (no jitter) & 51.4\% & Still underfitting, confirms mixup is too much for architecture \\
      v2.10 --- LR scheduling & --- & No major effects other than increasing training time \\
      v2.10b --- AutoContrast & 70.7\% & Insufficient effect on degraded set \\
      v2.13 --- Random erasing & 69.4\% & Black patches cannot approximate color crushing. \\
      \bottomrule
    \end{tabular}}
  \end{center}
  \vspace{0.5\baselineskip}
  Common pattern: each technique models a degradation \textit{not present} in the actual
  test set, adding training noise without improving robustness to real distortions.
\end{frame}

\begin{frame}{Iteration v2.7 \textendash\ Label Smoothing}
  We added \textbf{label smoothing} ($\varepsilon = 0.1$) to the cross-entropy loss.
  The rationale: with color distortion at test time, the model should not be
  overconfident on color-specific cues (e.g.\ sky / grass colors) that are unreliable on the
  degraded set.
  \begin{columns}[onlytextwidth]
    \begin{column}{0.35\linewidth}
      \begin{itemize}
        \item [train] Loss: 1.4173,\\ Acc: 0.8775
        \item [val] Loss: 1.5442,\\ Acc: 0.8100
        \item [clean] \textit{Acc}: 81.7\%,\\ \textit{F1}: 0.809
        \item [degraded] \textit{Acc}: 69.7\%,\\ \textit{F1}: 0.692
      \end{itemize}
    \end{column}
    \begin{column}{0.65\linewidth}
      \begin{center}
        \includegraphics[width=\textwidth]{../models/v2.7_label_smoothing.png}
      \end{center}
    \end{column}
  \end{columns}
  \vspace{0.5\baselineskip}
  Modest improvement in degraded robustness.
  Label smoothing was retained in subsequent runs as a regulariser alongside color jitter.
  \tiny\let\thefootnote\relax\footnotetext{Trained on RX6950 \textendash\ 16mins 41secs}
\end{frame}

\begin{frame}{Iteration v2.11 \textendash\ Batch Size}
  After optimizing \textbf{compute offloading} and using \textbf{Automatic Mixed Precision} for GPU computation,
  we decided to increase batch size $(32 \rightarrow 128)$ to take advantage of the improved pipeline,
  This didn't affect the network training too much and performance metrics stayed relatively stable.
  \vspace{\baselineskip}
  
  \begin{columns}[onlytextwidth]
    \begin{column}{0.3\linewidth}
      {\small\setlength{\leftmargin}{4em}
      \begin{itemize}
        \item [train] \textit{Acc}: 77.0\%,
        \item [val] \textit{Acc}: 78.0\%,
        \item [clean] \textit{Acc}: 81.9\%,\\ \textit{F1}: 0.8070
        \item [degraded] \textit{Acc}: 67.2\%,\\ \textit{F1}: 0.6673
      \end{itemize}}
    \end{column}
    \begin{column}{0.7\linewidth}
      \includegraphics[width=\textwidth]{../models/v2.11_batch_size_128.png}
    \end{column}
  \end{columns}
  \let\thefootnote\relax\footnotetext{Trained on RX9070 \textendash\ 28mins 29secs}
\end{frame}

\begin{frame}{Iteration v2.12 \textendash\ Scaling Up the Model}
  We scaled up from \textbf{Baseline2} to \textbf{Baseline3}
  (channel widths 32--512 $\to$ 64--512) and added
  \textbf{Test-Time Augmentation} (TTA:\@ 4 views averaged at inference):
  \vspace{\baselineskip}
  \begin{columns}[onlytextwidth]
    \begin{column}{0.3\linewidth}
      {\small\setlength{\leftmargin}{4em}
      \begin{itemize}
        \item [clean] \textit{Acc}: 85.6\%,\\ \textit{F1}: 0.852
        \item [degraded] \textit{Acc}: 73.4\%,\\ \textit{F1}: 0.731
        \item [deg.+TTA] \textit{Acc}: \textbf{74.4\%},\\ \textit{F1}: \textbf{0.749}
      \end{itemize}}
    \end{column}
    \begin{column}{0.7\linewidth}
      \includegraphics[width=\linewidth]{../models/v2.12_scale_up_model.png}
    \end{column}
  \end{columns}
  \let\thefootnote\relax\footnotetext{Trained on RX9070 \textendash\ 35mins 44secs}
\end{frame}

\subsection{Results}
\begin{frame}{Task 2 \textendash\ Results Summary and Conclusions}
  \begin{center}
    \begin{tabular}{lcc}
      \toprule
      Model / Strategy & Degraded Acc & F1 \\
      \midrule
      v1.12 (Task~1 model, no augment) & 56.2\% & 0.554 \\
      Best unsupervised enhancement (WB) & 53.2\% & 0.530 \\
      v2.0 --- ColorJitter (60 ep) & 58.6\% & 0.579 \\
      v2.2 --- Conservative params & 66.9\% & 0.668 \\
      v2.3 --- Analysis-driven params & 73.1\% & 0.727 \\
      v2.7 --- + Label smoothing & 69.7\% & 0.692 \\
      \textbf{v2.12 --- Baseline3 + TTA} & \textbf{74.4\%} & \textbf{0.749} \\
      \bottomrule
    \end{tabular}
  \end{center}
  \vspace{0.5\baselineskip}
  Conclusions:
  \begin{itemize}
    \item Unsupervised enhancement consistently \textit{worsened} results
    \item Augmentation-aware training closed the gap by \textbf{18 percentage points}
    \item Data-driven parameter selection (v2.3) was the single most impactful step ($+$6.2\%)
    \item Scaling up the model and TTA gave a further $+$1.3\% on top
  \end{itemize}
\end{frame}
